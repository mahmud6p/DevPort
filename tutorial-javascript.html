Build a To-Do App from Scratch

⏱ 45 min 🟢 Beginner 📁 HTML, CSS, JavaScript

What You Will Build

একটি সম্পূর্ণ To-Do App যেখানে task add, complete mark এবং delete করা যাবে। এই project এ তুমি JavaScript এর DOM manipulation, event handling এবং localStorage ব্যবহার শিখবে।

Step 3: JavaScript দিয়ে Functionality যোগ করো

app.js file তৈরি করো:

app.js
// Page load হলে saved tasks দেখাও
window.onload = function() {
  loadTasks();
}

// Task add করার function
function addTask() {
  const input = document.getElementById('taskInput');
  const taskText = input.value.trim();

  if (taskText === '') {
    alert('Please enter a task!');
    return;
  }

  createTaskElement(taskText, false);
  saveTask(taskText);
  input.value = '';
}

// Task element তৈরি করার function
function createTaskElement(text, isDone) {
  const ul = document.getElementById('taskList');
  const li = document.createElement('li');
  if (isDone) li.classList.add('done');

  li.innerHTML = `
    <span onclick="toggleDone(this)">${text}</span>
    <button class="delete-btn" onclick="deleteTask(this)">Delete</button>
  `;

  ul.appendChild(li);
}

// Task complete/incomplete toggle
function toggleDone(span) {
  span.parentElement.classList.toggle('done');
  updateStorage();
}

// Task delete করা
function deleteTask(btn) {
  btn.parentElement.remove();
  updateStorage();
}

// localStorage এ save করা
function saveTask(text) {
  let tasks = JSON.parse(localStorage.getItem('tasks')) || [];
  tasks.push({ text: text, done: false });
  localStorage.setItem('tasks', JSON.stringify(tasks));
}

// localStorage থেকে load করা
function loadTasks() {
  let tasks = JSON.parse(localStorage.getItem('tasks')) || [];
  tasks.forEach(task => {
    createTaskElement(task.text, task.done);
  });
}

// Storage update করা
function updateStorage() {
  const items = document.querySelectorAll('#taskList li');
  let tasks = [];
  items.forEach(item => {
    tasks.push({
      text: item.querySelector('span').textContent,
      done: item.classList.contains('done')
    });
  });
  localStorage.setItem('tasks', JSON.stringify(tasks));
}

// Enter key দিয়েও task add করা
document.getElementById('taskInput').addEventListener('keypress', function(e) {
  if (e.key === 'Enter') addTask();
});

💡 Tips: localStorage ব্যবহার করায় browser refresh করলেও tasks হারিয়ে যাবে না।

Step 4: Browser এ Test করো

তিনটি file একই folder এ রাখো এবং index.html browser এ open করো। Task লিখে Add button বা Enter চাপো — কাজ করবে!

কী কী শিখলে?

  • DOM manipulation (createElement, appendChild, remove)
  • Event handling (onclick, addEventListener)
  • localStorage দিয়ে data save ও load
  • Template literals ব্যবহার
  • JSON.parse এবং JSON.stringify
